28  import java.util.Map;

31  import com.sun.tracing.Provider;
32  import com.sun.tracing.ProviderFactory;

34  /**
35   * Factory class to create JSDT Providers.
36   *
37   * This class contains methods to create an instance of a Provider
38   * interface which can be used to place tracepoints in an application.
39   * Method calls upon that instance trigger DTrace probes that
40   * are visible from DTrace scripts.   Such calls have no other
41   * side effects in the application.
42   * <p>
43   * The DTrace script mechanisms for listing and matching probes will not see
44   * nor match any probes until the provider they reside in is created by a
45   * call to {@code createProvider()} (or {@code createProviders()}).
46   * <p>
47   * Providers that are created should be disposed of when they are no longer
48   * needed to free up system resources, at which point the associated
49   * DTrace probes will no longer be available to DTrace.  One disposes a
50   * provider by calling
51   * {@link com.sun.tracing.Provider#dispose Provider.dispose()} on a
52   * created provider instance.
53   *
54   * @since 1.7
55   */
56  public final class DTraceProviderFactory extends ProviderFactory {
57      /**
58       * Creates an instance of a provider which can then be used to trigger
59       * DTrace probes.
60       *
61       * The provider specification, provided as an argument, should only
62       * contain methods which have a 'void' return type and String or
63       * integer-based typed arguments (long, int, short, char, byte, or boolean).
64       *
65       * @param cls A user-defined interface which extends {@code Provider}.
66       * @return An instance of the interface which is used to trigger
67       * the DTrace probes.
68       * @throws java.lang.SecurityException if a security manager has been
69       * installed and it denies
70       * RuntimePermission("com.sun.dtrace.jsdt.createProvider")
71       * @throws java.lang.IllegalArgumentException if the interface contains
72       * methods that do not return null, or that contain arguments that are
73       * not String or integer types.
74       */
75      public <T extends Provider> T createProvider(Class<T> cls) {
76        throw unsupportedJSDTCompat();
77      }
78  
79    
80      /**
81       * Creates multiple providers at once.
82       *
83       * This method batches together a number of provider instantiations.
84       * It works similarly
85       * to {@code createProvider}, but operates on a set of providers instead
86       * of one at a time.  This method is in place since some DTrace
87       * implementations limit the number of times that providers can be
88       * created.  When numerous providers can be created at once with this
89       * method, it will count only as a single creation point to DTrace, thus
90       * it uses less system resources.
91       * <p>
92       * All of the probes in the providers will be visible to DTrace after
93       * this call and all will remain visible until all of the providers
94       * are disposed.
95       * <p>
96       * The {@code moduleName} parameter will override any {@code ModuleName}
97       * annotation associated with any of the providers in the set.
98       * All of the probes created by this call will share the same
99       * module name.
100      * <p>
101      * @param providers a set of provider specification interfaces
102      * @param moduleName the module name to associate with all probes
103      * @return A map which maps the provider interface specification to an
104      * implementing instance.
105      * @throws java.lang.SecurityException if a security manager has been
106      * installed and it denies
107      * RuntimePermission("com.sun.dtrace.jsdt.createProvider")
108      * @throws java.lang.IllegalArgumentException if any of the interface
109      * contains methods that do not return null, or that contain arguments
110      * that are not String or integer types.
111      */
112     public Map<Class<? extends Provider>,Provider> createProviders(
113             Set<Class<? extends Provider>> providers, String moduleName) {
114        throw unsupportedJSDTCompat();
115     }

117     /**
118      * Used to check the status of DTrace support in the underlying JVM and
119      * operating system.
120      *
121      * This is an informative method only - the Java-level effects of
122      * creating providers and triggering probes will not change whether or
123      * not DTrace is supported by the underlying systems.
124      *
125      * @return true if DTrace is supported
126      */
127     public static boolean isSupported() {
128         return false;
129     }

131     private final UnsupportedOperationException unsupportedJSDTCompat() {
132       throw new UnsupportedOperationException("not supported by JSDT-compat, remove JSDT-compat from classpath and use JDK 7 or later");
133     }
