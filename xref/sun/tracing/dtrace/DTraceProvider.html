91      /**
92       * Magic routine which creates an implementation of the user's interface.
93       *
94       * This method uses the ProxyGenerator directly to bypass the
95       * java.lang.reflect.proxy cache so that we get a unique class each
96       * time it's called and can't accidently reuse a $Proxy class.
97       *
98       * @return an implementation of the user's interface
99       */
100     @SuppressWarnings("unchecked")
101     public <T extends Provider> T newProxyInstance() {
102         /*
103          * Choose a name for the proxy class to generate.
104          */
105         long num = getUniqueNumber();
106 
107         String proxyPkg = "";
108         if (!Modifier.isPublic(providerType.getModifiers())) {
109             String name = providerType.getName();
110             int n = name.lastIndexOf('.');
111             proxyPkg = ((n == -1) ? "" : name.substring(0, n + 1));
112         }
113 
114         String proxyName = proxyPkg + proxyClassNamePrefix + num;
115 
116         /*
117          * Generate the specified proxy class.
118          */
119         Class<?> proxyClass = null;
120         byte[] proxyClassFile = ProxyGenerator.generateProxyClass(
121                 proxyName, new Class<?>[] { providerType });
122         try {
123             proxyClass = JVM.defineClass(
124                 providerType.getClassLoader(), proxyName,
125                 proxyClassFile, 0, proxyClassFile.length);
126         } catch (ClassFormatError e) {
127             /*
128              * A ClassFormatError here means that (barring bugs in the
129              * proxy class generation code) there was some other
130              * invalid aspect of the arguments supplied to the proxy
131              * class creation (such as virtual machine limitations
132              * exceeded).
133              */
134             throw new IllegalArgumentException(e.toString());
135         }
136 
137         /*
138          * Invoke its constructor with the designated invocation handler.
139          */
140         try {
141             Constructor cons = proxyClass.getConstructor(constructorParams);
142             return (T)cons.newInstance(new Object[] { this });
143         } catch (NoSuchMethodException e) {
144             throw new InternalError(e.toString());
145         } catch (IllegalAccessException e) {
146             throw new InternalError(e.toString());
147         } catch (InstantiationException e) {
148             throw new InternalError(e.toString());
149         } catch (InvocationTargetException e) {
150             throw new InternalError(e.toString());
151         }
152     }
153 
154     // In the normal case, the proxy object's method implementations will call
155     // this method (it usually calls the ProviderSkeleton's version).  That
156     // method uses the passed 'method' object to lookup the associated
157     // 'ProbeSkeleton' and calls uncheckedTrigger() on that probe to cause the
158     // probe to fire.  DTrace probes are different in that the proxy class's
159     // methods are immediately overridden with native code to fire the probe
160     // directly.  So this method should never get invoked.  We also wire up the
161     // DTraceProbe.uncheckedTrigger() method to call the proxy method instead
162     // of doing the work itself.
163     public Object invoke(Object proxy, Method method, Object[] args) {
164         if (method.getDeclaringClass() != providerType) {
165             try {
166                 return method.invoke(this, args);
167             } catch (IllegalAccessException e) {
168                 assert false;
169             } catch (InvocationTargetException e) {
170                 assert false;
171             }
172         } else if (active) {
173             assert false : "This method should have been overridden by the JVM";
174         }
175         return null;
176     }
