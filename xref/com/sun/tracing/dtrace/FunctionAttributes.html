28  import java.lang.annotation.Target;

31  import java.lang.annotation.ElementType;

33  /**
34   * This annotation describes the interface attributes of the
35   * {@code function} field for a single provider.
36   *
37   * This annotation can be added to a user-defined {@code Provider} specification
38   * interface to set the stability attributes of the {@code function} field for
39   * all probes specified in that provider.
40   * <p>
41   * If this annotation is not present, the interface attributes for the
42   * {@code function} field are Private/Private/Unknown.
43   * <p>
44   * @see <a href="http://docs.sun.com/app/docs/doc/817-6223/6mlkidlnp?a=view">Solaris Dynamic Tracing Guide, Chapter 39: Stability</a>
45   * @since 1.7
46   */
47  @Retention(RetentionPolicy.RUNTIME)
48  @Target({ ElementType.TYPE })
49  public @interface FunctionAttributes {
50      Attributes value();
51  }
